api-integration.js
const AgentForge = require('agentforge-sdk');

const codeReviewAgent = new AgentForge.Agent({
    name: 'code-reviewer',
    capabilities: ['github', 'openai'],
    triggers: {
        pullRequest: true,
        schedule: 'daily'
    },
    actions: [
        {
            type: 'reviewCode',
            params: {
                strictness: 'high',
                language: 'javascript'
            }
        },
        {
            type: 'generateReport',
            format: 'markdown'
        }
    ]
});

codeReviewAgent.deploy();
